@app.route("/influ_register", methods=["GET", "POST"])
def influ_register():
    if request.method=="GET":
        return render_template("influ_register.html")
    if request.method=="POST":
        name=request.form.get("name")
        password=request.form.get("password")     
        try:
            influencer_from_db=Influencer.query.get(name)
            if influencer_from_db:
                password_from_db=influencer_from_db.password
                if password_from_db==password:
                    global logged_influencer
                    logged_influencer=name
                    return "Influencer"
                else:
                    return render_template("influ_register.html", message="Password failed")
            else:
                return render_template("influ_register.html",message="id failed")
        except:
            return "some error"
